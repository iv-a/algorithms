function solution(requiredAmount, numberOfDenominations, denominations) {
  const dp = new Array(requiredAmount + 1).fill(Infinity);

  for (let i = 1; i <= numberOfDenominations; i++) {
    for (let j = 1; j <= requiredAmount; j++) {
