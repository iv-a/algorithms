  // –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Å–∞–º—É—é –¥–ª–∏–Ω–Ω—É—é –∏ —Å–∞–º—É—é –∫–æ—Ä–æ—Ç–∫—É—é —Å—Ç—Ä–æ–∫–∏

  for (let i = 0; i <= longestString.length; i++) {
    for (let j = 0; j <= shortestString.length; j++) {
      if (i === 0 && j === 0) {
        // –ï—Å–ª–∏ –æ–±–µ —Å—Ç—Ä–æ–∫–∏ –ø—É—Å—Ç—ã–µ, —Ç–æ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –õ–µ–≤–µ–Ω—à—Ç–µ–π–Ω–∞ —Ä–∞–≤–Ω–æ –Ω—É–ª—é

        // –ï—Å–ª–∏ –±–û–ª—å—à–∞—è —Å—Ç—Ä–æ–∫–∞ –ø—É—Å—Ç–∞—è, —Ç–æ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –õ–µ–≤–µ–Ω—à—Ç–µ–π–Ω–∞ —Ä–∞–≤–Ω–æ –¥–ª–∏–Ω–µ –º–µ–Ω—å—à–µ–π —Å—Ç—Ä–æ–∫–∏

      } else if (j === 0 && i > 0) {
        // –ï—Å–ª–∏ –º–µ–Ω—å—à–∞—è —Å—Ç—Ä–æ–∫–∞ –ø—É—Å—Ç–∞—è, —Ç–æ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –õ–µ–≤–µ–Ω—à—Ç–µ–π–Ω–∞ —Ä–∞–≤–Ω–æ –¥–ª–∏–Ω–µ –±–æ–ª—å—à–µ–π —Å—Ç—Ä–æ–∫–∏

      } else if (longestString[i - 1] === shortestString[j - 1]) {
        // –ï—Å–ª–∏ —Å–∏–º–≤–æ–ª—ã —Ä–∞–≤–Ω—ã, —Ç–æ —Å–æ–≤–µ—Ä—à–∞—Ç—å –∞—Ç–æ–º–∞—Ä–Ω—ã–µ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è –Ω–µ —Ç—Ä–µ–±—É–µ—Ç—Å—è,
        // –∑–Ω–∞—á–∏—Ç —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –õ–µ–≤–µ–Ω—à—Ç–µ–π–Ω–∞ –Ω–µ –∏–∑–º–µ–Ω–∏–ª–æ—Å—å

      } else if (i > 0 && j > 0 && longestString[i - 1] !== shortestString[j - 1]) {
        // –ï—Å–ª–∏ —Å—Ç—Ä–æ–∫–∏ –Ω–µ –Ω—É–ª–µ–≤—ã–µ –∏ –∏—Ö —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–µ —Å–∏–º–≤–æ–ª—ã –Ω–µ —Ä–∞–≤–Ω—ã, —Ç–æ
        // –†–∞—Å—Å—Ç–æ—è–Ω–∏–µ –õ–µ–≤–µ–Ω—à—Ç–µ–π–Ω–∞ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è –º–∏–Ω–∏–º–∞–ª—å–Ω—ã–º –∏–∑ —Ç—Ä–µ—Ö –∑–Ω–∞—á–µ–Ω–∏–π
