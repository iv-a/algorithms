function solution(numberOfWins, scores) {
  let sum = 0;
  // –í—ã—á–∏—Å–ª—è–µ–º –∑–Ω–∞—á–µ–Ω–∏–µ —Å—É–º–º—ã –≤—Å–µ—Ö –Ω–∞–±—Ä–∞–Ω–Ω—ã—Ö –æ—á–∫–æ–≤

    sum += score;

  // –ü—Ä–æ–≤–µ—Ä—è–µ–º —Å—É–º–º—É –Ω–∞ —á–µ—Ç–Ω–æ—Å—Ç—å
  if (sum % 2 === 1) {
    return CONSTANTS.ANS.FALSE;

  // –í—ã—á–∏—Å–ª—è–µ–º –ø–æ–ª–æ–≤–∏–Ω—É —Å—É–º–º—ã
  const halfOfSum = sum >> 1;
  // –°–æ–∑–¥–∞–µ–º –º–∞—Å—Å–∏–≤ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è —Ç–µ–∫—É—â–∏—Ö –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–∏—Ö —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤

  // –°–æ–∑–¥–∞–µ–º –º–∞—Å—Å–∏–≤ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –ø—Ä–µ–¥—ã–¥—É—â–∏—Ö –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–∏—Ö —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
  let previousDP = new Array(halfOfSum + 1);

  for (let i = 0; i <= numberOfWins; i++) {
    for (let j = 0; j <= halfOfSum; j++) {
      if (i === 0 && j > 0) {
        // –í –ø—É—Å—Ç–æ–º –º–Ω–æ–∂–µ—Å—Ç–≤–µ –Ω–µ –º–æ–∂–µ—Ç –±—ã—Ç—å –Ω–µ –ø—É—Å—Ç–æ–≥–æ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–∞
        currentDP[j] = false;
      } else if (j === 0 && i >= 0) {
        // –í –ª—é–±–æ–º –º–Ω–æ–∂–µ—Å—Ç–≤–µ –º–æ–∂–µ—Ç –±—ã—Ç—å –ø—É—Å—Ç–æ–µ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ
        currentDP[j] = true;
      } else {
        if (j - scores[i - 1] >= 0) {
          currentDP[j] = previousDP[j - scores[i - 1]] || previousDP[j];
        } else {
          currentDP[j] = previousDP[j];
        }
      }
    }
    // –ü–æ—Å–∫–æ–ª—å–∫—É –¥–ª—è –≤—ã—á–∏—Å–ª–µ–Ω–∏—è currentDP —Ç—Ä–µ–±—É–µ—Ç—Å—è —Ö—Ä–∞–Ω–∏—Ç—å —Ç–æ–ª—å–∫–æ previousDP, —Ç–æ
    // –≤ –∫–æ–Ω—Ü–µ —Ü–∏–∫–ª–∞ —Å–æ—Ö—Ä–∞–Ω—è–µ–º —Ç–µ–∫—â–∏–µ –ø—Ä–æ–º–µ–∂—É—Ç–æ—á–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è –≤ –∫–∞—á–µ—Å—Ç–≤–µ –ø—Ä–µ–¥—ã–¥—É—â–∏—Ö
    [ previousDP, currentDP ] = [ currentDP, previousDP ]
  }
  // –í–æ–∑–≤—Ä–∞—â–∞–µ–º –æ—Ç–≤–µ—Ç –∫ –∑–∞–¥–∞—á–µ
  return previousDP[halfOfSum] ? CONSTANTS.ANS.TRUE : CONSTANTS.ANS.FALSE;
}

const io_interface = readline.createInterface({input: process.stdin});

};

  } else if (lineNumber === 1) {
    scores = line.split(' ').map((item) => parseInt(item, DEX));
  }
  lineNumber++;
});
