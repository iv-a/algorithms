  –ú - —Å—Ä–µ–¥–Ω—è—è –¥–ª–∏–Ω–∞ —Ä–∞—Å–ø–∞–∫–æ–≤–∞–Ω–Ω–æ–π —Å—Ç—Ä–æ–∫–∏;

function isNumber(char) {
  return char >= '0' && char <= '9';
}

// –§—É–Ω–∫—Ü–∏—è —Ä–∞—Å–ø–∞–∫–æ–≤–∫–∏ —Å—Ç—Ä–æ–∫–∏
function unpack(string) {
  // –ó–∞–≤–æ–¥–∏–º —Å—Ç–µ–∫ –¥–ª—è –¥–ª—è –∑–∞–ø–∞–∫–æ–≤–∞–Ω–Ω—ã—Ö —Å—Ç—Ä–æ–∫ (–ó–°)
  const stack = [];
  // –ó–∞–≤–æ–¥–∏–º —Å—Ç–µ–∫ –¥–ª—è –º–Ω–æ–∂–∏—Ç–µ–ª–µ–π

  // –í —Ü–∏–∫–ª–µ –ø—Ä–æ—Ö–æ–¥–∏–º—Å—è –ø–æ –∫–∞–∂–¥–æ–º—É —Å–∏–º–≤–æ–ª—É —Å—Ç—Ä–æ–∫–∏
  for (let i = 0; i < string.length; i++) {
    const char = string[i];
    if (isNumber(char)) {
      // –ï—Å–ª–∏ —Å–∏–º–≤–æ–ª —è–≤–ª—è–µ—Ç—Å—è —Ü–∏—Ñ—Ä–æ–π, —Ç–æ –¥–æ–±–∞–≤–ª—è–µ–º –µ–≥–æ –≤ —Å—Ç–µ–∫ –º–Ω–æ–∂–∏—Ç–µ–ª–µ–π
      multipliers.push(parseInt(char, DEX));
    } else if (char === BRACKET.OPEN) {
      // –ï—Å–ª–∏ —Å–∏–º–≤–æ–ª —è–≤–ª—è–µ—Ç—Å—è –æ—Ç–∫—Ä—ã–≤–∞—é—â–µ–π —Å–∫–æ–±–∫–æ–π, —Ç–æ –¥–æ–±–∞–≤–ª—è–µ–º –≤ —Å—Ç–µ–∫ –ø—É—Å—Ç—É—é —Å—Ç—Ä–æ–∫—É
      stack.push('');
    } else if (char === BRACKET.CLOSE) {
      // –ï—Å–ª–∏ —Å–∏–º–≤–æ–ª —è–≤–ª—è–µ—Ç—Å—è –∑–∞–∫—Ä—ã–≤–∞—é—â–µ–π —Å–∫–æ–±–∫–æ–π, —Ç–æ –∏–∑–≤–ª–µ–∫–∞–µ–º –º–Ω–æ–∂–∏—Ç–µ–ª—å —Å –≤–µ—Ä—à–∏–Ω—ã —Å—Ç–µ–∫–∞ –º–Ω–æ–∂–∏—Ç–µ–ª–µ–π
      const multiplier = multipliers.pop();
      // –ò–∑–≤–ª–µ–∫–∞–µ–º –ó–° —Å –≤–µ—Ä—à–∏–Ω—ã —Å—Ç–µ–∫–∞ –∏ –ø–æ–≤—Ç–æ—Ä—è–µ–º –µ–µ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–µ–µ —á–∏—Å–ª–æ —Ä–∞–∑
      let unpackedString = stack.pop().repeat(multiplier);
      // –ò–∑–≤–ª–µ–∫–∞–µ–º –∏–∑ —Å—Ç–µ–∫–∞ —Ä–∞—Å–ø–∞–∫–æ–≤–∞–Ω–Ω—É—é —Ä–∞–Ω–µ–µ —á–∞—Å—Ç—å —Å—Ç—Ä–æ–∫–∏ –∏ —Å—É–º–º–∏—Ä—É–µ–º —Å —Ç–µ–∫—É—â–µ–π —Ä–∞—Å–ø–∞–∫–æ–≤–∞–Ω–Ω–æ–π
      // –†–µ–∑—É–ª—å—Ç–∞—Ç –ø–æ–º–µ—â–∞–µ–º –æ–±—Ä–∞—Ç–Ω–æ –≤ —Å—Ç–µ–∫
      stack.push((stack.length ? stack.pop() : '') + unpackedString);
    } else {
      // –ò–∑–≤–ª–µ–∫–∞–µ–º –∏–∑ —Å—Ç–µ–∫–∞ –∑–∞–ø–∞–∫–æ–≤–∞–Ω–Ω—É—é —á–∞—Å—Ç—å —Å—Ç—Ä–æ–∫–∏ –∏ —Å—É–º–º–∏—Ä—É–µ–º —Å —Ç–µ–∫—É—â–∏–º —Å–∏–º–≤–æ–ª–æ–º
      // –†–µ–∑—É–ª—å—Ç–∞—Ç –ø–æ–º–µ—â–∞–µ–º –æ–±—Ä–∞—Ç–Ω–æ –≤ —Å—Ç–µ–∫
      stack.push((stack.length ? stack.pop() : '') + char);
    }
  }
  // –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Ä–∞—Å–ø–∞–∫–æ–≤–∞–Ω–Ω—É—é —Å—Ç—Ä–æ–∫—É
  return stack.pop();
}

// –§—É–Ω–∫—Ü–∏—è –ø–æ–∏—Å–∫–∞ –Ω–∞–∏–±–æ–ª—å—à–µ–≥–æ –æ–±—â–µ–≥–æ –ø—Ä–µ—Ñ–∏–∫—Å–∞
function getLCP(array, minLength) {
  // –ó–∞–≤–æ–¥–∏–º –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞
  let result = '';
  // –ü—Ä–æ—Ö–æ–¥–∏–º—Å—è –≤ —Ü–∏–∫–ª–µ –ø–æ –∫–∞–∂–¥–æ–º—É —Å–∏–º–≤–æ–ª—É –æ—Ç 0 –¥–æ –¥–ª–∏–Ω—ã –Ω–∞–∏–º–µ–Ω—å—à–µ–π —Å—Ç—Ä–æ–∫–∏
  for (let i = 0; i < minLength; i++) {
    // –ó–∞–ø–æ–º–∏–Ω–∞–µ–º —Å–∏–º–≤–æ–ª —Å–∞–º–æ–π –ø–µ—Ä–≤–æ–π —Å—Ç—Ä–æ–∫–∏
    let char = array[0][i];
    // –ü—Ä–æ—Ö–æ–¥–∏–º—Å—è –ø–æ –∫–∞–∂–¥–æ–π —Å—Ç—Ä–æ–∫–µ
    for (let j = 0; j < array.length; j++) {
      // –ï—Å–ª–∏ —Å–∏–º–≤–æ–ª —Ç–µ–∫—É—â–µ–π —Å—Ç—Ä–æ–∫–∏ –Ω–µ —Ä–∞–≤–µ–Ω —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–µ–º—É —Å–∏–º–≤–æ–ª—É —Å–∞–º–æ–π –ø–µ—Ä–≤–æ–π —Å—Ç—Ä–æ–∫–∏,
      // —Ç–æ –≤–æ–∑–≤—Ä–∞—â–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç
      if (char !== array[j][i]) {
        return result;
      }
    }
    // –ï—Å–ª–∏ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–µ–µ —Å–∏–º–≤–æ–ª—ã –≤—Å–µ—Ö —Å—Ç—Ä–æ–∫ —Ä–∞–≤–Ω—ã, –¥–æ–±–∞–≤–ª—è–µ–º —ç—Ç–æ—Ç —Å–∏–º–≤–æ–ª –∫ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—É
    result += char;
  }
  // –í–æ–∑–≤—Ä–∞—â–∞–µ–º –Ω–∞–∏–±–æ–ª—å—à–∏–π –æ–±—â–∏–π –ø—Ä–µ—Ñ–∏–∫—Å
  return result;
}
